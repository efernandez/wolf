134 void CeresManager::update(WolfProblem* _problem_ptr)
135 {
136  // IF REALLOCATION OF STATE, REMOVE EVERYTHING AND BUILD THE PROBLEM AGAIN
137  if (_problem_ptr->isReallocated())
138  {
139  // Remove all parameter blocks (residual blocks will be also removed)
140  removeAllStateUnits();
141 
142  // Add all parameter blocks
143  for(auto state_unit_it = _problem_ptr->getStateListPtr()->begin(); state_unit_it!=_problem_ptr->getStateListPtr()->end(); state_unit_it++)
144  addStateUnit(*state_unit_it);
145 
146  // Add all residual blocks
147  ConstraintBaseList ctr_list;
148  _problem_ptr->getTrajectoryPtr()->getConstraintList(ctr_list);
149  for(auto ctr_it = ctr_list.begin(); ctr_it!=ctr_list.end(); ctr_it++)
150  addConstraint(*ctr_it);
151 
152  // set the wolf problem reallocation flag to false
153  _problem_ptr->reallocationDone();
154  }
155  else
156  {
157  // ADD/UPDATE STATE UNITS
158  for(auto state_unit_it = _problem_ptr->getStateListPtr()->begin(); state_unit_it!=_problem_ptr->getStateListPtr()->end(); state_unit_it++)
159  {
160  if ((*state_unit_it)->getPendingStatus() == ADD_PENDING)
161  addStateUnit(*state_unit_it);
162 
163  else if((*state_unit_it)->getPendingStatus() == UPDATE_PENDING)
164  updateStateUnitStatus(*state_unit_it);
165  }
166  //std::cout << "state units updated!" << std::endl;
167 
168  // REMOVE STATE UNITS
169  while (!_problem_ptr->getRemovedStateListPtr()->empty())
170  {
171  ceres_problem_->RemoveParameterBlock(_problem_ptr->getRemovedStateListPtr()->front());
172  _problem_ptr->getRemovedStateListPtr()->pop_front();
173  }
174  //std::cout << "state units removed!" << std::endl;
175 
176  // ADD CONSTRAINTS
177  ConstraintBaseList ctr_list;
178  _problem_ptr->getTrajectoryPtr()->getConstraintList(ctr_list);
179  //std::cout << "ctr_list.size() = " << ctr_list.size() << std::endl;
180  for(auto ctr_it = ctr_list.begin(); ctr_it!=ctr_list.end(); ctr_it++)
181  if ((*ctr_it)->getPendingStatus() == ADD_PENDING)

184  //std::cout << "constraints updated!" << std::endl;
