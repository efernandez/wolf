46 void motionCampus(unsigned int ii, Cpose3d & pose, double& displacement_, double& rotation_)
47 {
48  if (ii <= 120)
49  {
50  displacement_ = 0.1;
51  rotation_ = 0;
52  }
53  else if ((ii > 120) && (ii <= 170))
54  {
55  displacement_ = 0.2;
56  rotation_ = 1.8 * M_PI / 180;
57  }
58  else if ((ii > 170) && (ii <= 220))
59  {
60  displacement_ = 0;
61  rotation_ = -1.8 * M_PI / 180;
62  }
63  else if ((ii > 220) && (ii <= 310))
64  {
65  displacement_ = 0.1;
66  rotation_ = 0;
67  }
68  else if ((ii > 310) && (ii <= 487))
69  {
70  displacement_ = 0.1;
71  rotation_ = -1. * M_PI / 180;
72  }
73  else if ((ii > 487) && (ii <= 600))
74  {
75  displacement_ = 0.2;
76  rotation_ = 0;
77  }
78  else if ((ii > 600) && (ii <= 700))
79  {
80  displacement_ = 0.1;
81  rotation_ = -1. * M_PI / 180;
82  }
83  else if ((ii > 700) && (ii <= 780))
84  {
85  displacement_ = 0;
86  rotation_ = -1. * M_PI / 180;
87  }
88  else
89  {
90  displacement_ = 0.3;
91  rotation_ = 0.0 * M_PI / 180;
92  }
93 
94  pose.moveForward(displacement_);
95  pose.rt.setEuler(pose.rt.head() + rotation_, pose.rt.pitch(), pose.rt.roll());
96 }
