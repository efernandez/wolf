130 void EpipolarConstraint::computeExpectation()
131 {
132 
133  // Test if we have TransSensor to the other Capture and get it by its node ID.
134  auto trans_sensor_ptr = capturePtr()->transSensorPtr( captureOtherPtr()->nodeId() );
135 
136  if (trans_sensor_ptr != NULL) // TRANS-SENSOR FOUND: USE ALL PRE-COMPUTED DATA
137  {
138  // Delegate all the work to the TransSensor
139  trans_sensor_ptr->computeExpectation(featureOtherPtr()->measurement(), expectation_);
140  }
141  else // NO TRANS-SENSOR FOUND: RE-COMPUTE RELATIVE POSE AND FUNDAMENTAL MATRIX.
142  {
143 
144  using namespace Eigen;
145 
146  // NOTE: We use a temporary "Matrix3s M_tmp;" that will be used several times
147 
148  // 1. Start with the poses stuff:
149 
150  // 1a. Composition of the relative pose - this is because we do not have trans-sensor, so we need to compute them each time.
151  // We make use of pre-computed global poses in Capture.
152  // M_tmp is R_rel, the relative rotation matrix:
153  Matrix3s M_tmp = ( captureOtherPtr()->inverseGlobalPose().q() * capturePtr()->globalPose().q() ).matrix();
154  Vector3s T_rel = captureOtherPtr()->inverseGlobalPose().q() * capturePtr()->globalPose().p()
155  + captureOtherPtr()->inverseGlobalPose().p();
156 
157  // 2. Now on to the vision stuff...
158 
159  // 2a. Cast SensorBase to PinHole to access the inverse intrinsic matrices, pre-computed in PinHole at construction time.
160  // NOTE: A dynamic cast is safer but here we vote for speed.
161  PinHole* pin_hole_own_ptr_(static_cast<PinHolePtr>(sensorPtr() ) );
162  PinHole* pin_hole_other_ptr_(static_cast<PinHolePtr>(sensorOtherPtr() ) );
163 
164  // 2b. Compute essential and fundamental matrices - this is because we do not have trans-sensor, so we need to compute them each time.
165  // helper: E = R_rel' * [T_rel]_x
166  // helper: F = (K_own^-1)' * E * K_other^-1
167  // M_tmp becomes E, the essential matrix:
168  M_tmp = M_tmp.transpose() * Wolf::skew(T_rel);
169  // M_tmp becomes F, the fundamental matrix:
170  M_tmp = pin_hole_own_ptr_->inverseIntrinsicMatrix().transpose() * M_tmp * pin_hole_other_ptr_->inverseIntrinsicMatrix();
171 
172  // 2c. compute the epipolar line
173  // This is equivalent to epi_line = F * euclideanToHomogeneous(pix_other);
174  expectation_ = M_tmp.leftCols(2) * featureOtherPtr()->measurement() + M_tmp.rightCols(1);
175  }
176 }
