146 void WolfManager<StatePositionT, StateOrientationT, StateVelocityT, StateOmegaT>::createFrame(const Eigen::VectorXs& _frame_state, const TimeStamp& _time_stamp)
147 {
148  //std::cout << "creating new frame..." << std::endl;
149 
150  assert(_frame_state.size() == StatePositionT::BLOCK_SIZE + StateOrientationT::BLOCK_SIZE + StateVelocityT::BLOCK_SIZE + StateOmegaT::BLOCK_SIZE && "Wrong frame vector when creating a new frame");
151 
152  // Process current frame non-odometry captures
153  if (current_frame_ != nullptr)
154  {
155  //std::cout << "Processing last frame non-odometry captures " << current_frame_->getCaptureListPtr()->size() << std::endl;
156  for (auto capture_it = current_frame_->getCaptureListPtr()->begin(); capture_it != current_frame_->getCaptureListPtr()->end(); capture_it++)
157  if ((*capture_it)->getSensorPtr() != sensor_prior_)
158  {
159  //std::cout << "processing capture " << (*capture_it)->nodeId() << std::endl;
160  (*capture_it)->processCapture();
161  }
162  }
163  //std::cout << "Last frame non-odometry captures processed" << std::endl;
164 
165  // Store last frame
166  previous_frame_ = current_frame_;
167 
168  // Create frame and add it to the trajectory
169  StatePositionT* new_position = new StatePositionT(problem_->getNewStatePtr());
170  problem_->addState(new_position, _frame_state.head<StatePositionT::BLOCK_SIZE>());
171  //std::cout << "StatePosition created" << std::endl;
172 
173  StateOrientationT* new_orientation = new StateOrientationT(problem_->getNewStatePtr());
174  problem_->addState(new_orientation, _frame_state.segment<StateOrientationT::BLOCK_SIZE>(new_position->getStateSize()));
175  //std::cout << "StateOrientation created" << std::endl;
176 
177  StateVelocityT* new_velocity = nullptr;
178  if (StateVelocityT::BLOCK_SIZE != 0)
179  {
180  new_velocity = new StateVelocityT(problem_->getNewStatePtr());
181  problem_->addState(new_velocity, _frame_state.segment<StateVelocityT::BLOCK_SIZE>(new_position->getStateSize()+new_orientation->getStateSize()));
182  }
183 
184  StateOmegaT* new_omega = nullptr;
185  if (StateOmegaT::BLOCK_SIZE != 0)
186  {
187  new_omega = new StateOmegaT(problem_->getNewStatePtr());
188  problem_->addState(new_omega, _frame_state.segment<StateOmegaT::BLOCK_SIZE>(new_position->getStateSize()+new_orientation->getStateSize()+new_velocity->getStateSize()));
189  }
190 
191  problem_->getTrajectoryPtr()->addFrame(new FrameBase(_time_stamp, new_position, new_orientation, new_velocity, new_omega));
192  //std::cout << "frame created" << std::endl;
193 
194  // Store new current frame
195  current_frame_ = problem_->getLastFramePtr();
196  //std::cout << "current_frame_" << std::endl;
197 
198  // empty last capture relative
199  if (previous_frame_ != nullptr)
200  {
201  CaptureRelative* empty_odom = new CaptureOdom2D(_time_stamp, _time_stamp, sensor_prior_, Eigen::Vector3s::Zero());
202  previous_frame_->addCapture(empty_odom);
203  empty_odom->processCapture();
204  last_capture_relative_ = empty_odom;
205  }
206  //std::cout << "last_frame_" << std::endl;
207 
208  // Fixing or removing old frames
209  manageWindow();
210  //std::cout << "new frame created" << std::endl;
211 }

253 template <class StatePositionT, class StateOrientationT, class StateVelocityT, class StateOmegaT>
254 void WolfManager<StatePositionT, StateOrientationT, StateVelocityT, StateOmegaT>::update()
255 {
256  //std::cout << "updating..." << std::endl;
257  while (!new_captures_.empty())
258  {
259  // EXTRACT NEW CAPTURE
260  CaptureBase* new_capture = new_captures_.front();
261  new_captures_.pop();
262 
263  // OVERWRITE CURRENT STAMP
264  current_frame_->setTimeStamp(new_capture->getTimeStamp());
265 
266  // INITIALIZE FIRST FRAME STAMP
267  if (previous_frame_->getTimeStamp().get() == 0)
268  previous_frame_->setTimeStamp(new_capture->getTimeStamp());
269 
270  // ODOMETRY SENSOR
271  if (new_capture->getSensorPtr() == sensor_prior_)
272  {
273  //std::cout << "adding odometry capture..." << new_capture->nodeId() << std::endl;
274  // NEW KEY FRAME ?
275  if (checkNewFrame(new_capture))
276  createFrame(new_capture->getTimeStamp());
277 
278  // ADD/INTEGRATE NEW ODOMETRY TO THE LAST FRAME
279  last_capture_relative_->integrateCapture((CaptureRelative*) (new_capture));
280  current_frame_->setState(last_capture_relative_->computePrior(new_capture->getTimeStamp()));
281  current_frame_->setTimeStamp(new_capture->getTimeStamp());
282  }
283  else
284  {
285  //std::cout << "adding not odometry capture..." << new_capture->nodeId() << std::endl;
286  // NEW KEY FRAME ?
287  if (checkNewFrame(new_capture))
288  createFrame(new_capture->getTimeStamp());
289 
290  // ADD CAPTURE TO THE CURRENT FRAME (or substitute the same sensor previous capture)
291  //std::cout << "searching repeated capture..." << new_capture->nodeId() << std::endl;
292  CaptureBaseIter repeated_capture_it = current_frame_->hasCaptureOf(new_capture->getSensorPtr());
293 
294  if (repeated_capture_it != current_frame_->getCaptureListPtr()->end()) // repeated capture
295  {
296  //std::cout << "repeated capture, keeping new capture" << new_capture->nodeId() << std::endl;
297  current_frame_->removeCapture(repeated_capture_it);
298  current_frame_->addCapture(new_capture);
299  }
300  else
301  {
302  //std::cout << "not repeated, adding capture..." << new_capture->nodeId() << std::endl;
303  current_frame_->addCapture(new_capture);
304  }
305  }
306  }
307  //std::cout << "updated" << std::endl;
308 }
