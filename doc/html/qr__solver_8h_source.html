293  bool solve(const unsigned int mode)
294  {
295  if (n_new_constraints_ == 0)
296  return 1;
297 
298  std::cout << "solving mode " << mode << std::endl;
299 
300  bool batch, order;
301  unsigned int first_ordered_idx;
302 
303  switch(mode)
304  {
305  case 0:
306  {
307  batch = true;
308  order = false;
309  break;
310  }
311  case 1:
312  {
313  batch = true;
314  order = (nNodes() > 1);
315  break;
316  }
317  case 2:
318  {
319  first_ordered_idx = findFirstOrderedNode();
320  batch = (nodeOrder(first_ordered_idx) == 0);
321  order = (nNodes() > 1);
322  }
323  }
324 
325  // BATCH
326  if (batch)
327  {
328  // REORDER
329  if (order)
330  ordering(-1);
331 
332  //printProblem();
333 
334  // SOLVE
335  t_solving_ = clock();
336  A_.makeCompressed();
337  solver_.compute(A_);
338  if (solver_.info() != Success)
339  {
340  std::cout << "decomposition failed" << std::endl;
341  return 0;
342  }
343  x_incr_ = solver_.solve(-b_);
344  R_ = solver_.matrixR();
345  //std::cout << "R" << std::endl << MatrixXd::Identity(R_.cols(), R_.cols()) * R_ << std::endl;
346  time_solving_ += ((double) clock() - t_solving_) / CLOCKS_PER_SEC;
347  }
348  // INCREMENTAL
349  else
350  {
351  // REORDER SUBPROBLEM
352  ordering(first_ordered_idx);
353  //printProblem();
354 
355  // SOLVE ORDERED SUBPROBLEM
356  t_solving_= clock();
357  A_nodes_.makeCompressed();
358  A_.makeCompressed();
359 
360  // finding measurements block
361  SparseMatrix<int> measurements_to_initial = A_nodes_.col(nodeOrder(first_ordered_idx));
362  //std::cout << "measurements_to_initial " << measurements_to_initial << std::endl;
363  //std::cout << "measurements_to_initial.innerIndexPtr()[measurements_to_initial.outerIndexPtr()[0]] " << measurements_to_initial.innerIndexPtr()[measurements_to_initial.outerIndexPtr()[0]] << std::endl;
364  unsigned int first_ordered_measurement = measurements_to_initial.innerIndexPtr()[measurements_to_initial.outerIndexPtr()[0]];
365  unsigned int ordered_measurements = A_.rows() - constraint_locations_.at(first_ordered_measurement);
366  unsigned int ordered_variables = A_.cols() - nodeLocation(first_ordered_idx);//nodes_.at(first_ordered_idx).location;
367  unsigned int unordered_variables = nodeLocation(first_ordered_idx);//nodes_.at(first_ordered_idx).location;
368 
369  SparseMatrix<double> A_partial = A_.bottomRightCorner(ordered_measurements, ordered_variables);
370  solver_.compute(A_partial);
371  if (solver_.info() != Success)
372  {
373  std::cout << "decomposition failed" << std::endl;
374  return 0;
375  }
376  //std::cout << "R new" << std::endl << MatrixXd::Identity(A_partial.cols(), A_partial.cols()) * solver_.matrixR() << std::endl;
377  x_incr_.tail(ordered_variables) = solver_.solve(-b_.tail(ordered_measurements));
378 
379  // store new part of R
380  eraseSparseBlock(R_, unordered_variables, unordered_variables, ordered_variables, ordered_variables);
381  //std::cout << "R" << std::endl << MatrixXd::Identity(R_.rows(), R_.rows()) * R_ << std::endl;
382  addSparseBlock(solver_.matrixR(), R_, unordered_variables, unordered_variables);
383  //std::cout << "R" << std::endl << MatrixXd::Identity(R_.rows(), R_.rows()) * R_ << std::endl;
384  R_.makeCompressed();
385 
386  // solving not ordered subproblem
387  if (unordered_variables > 0)
388  {
389  //std::cout << "--------------------- solving unordered part" << std::endl;
390  SparseMatrix<double> R1 = R_.topLeftCorner(unordered_variables, unordered_variables);
391  //std::cout << "R1" << std::endl << MatrixXd::Identity(R1.rows(), R1.rows()) * R1 << std::endl;
392  SparseMatrix<double> R2 = R_.topRightCorner(unordered_variables, ordered_variables);
393  //std::cout << "R2" << std::endl << MatrixXd::Identity(R2.rows(), R2.rows()) * R2 << std::endl;
394  solver_.compute(R1);
395  if (solver_.info() != Success)
396  {
397  std::cout << "decomposition failed" << std::endl;
398  return 0;
399  }
400  x_incr_.head(unordered_variables) = solver_.solve(-b_.head(unordered_variables) + R2 * x_incr_.tail(ordered_variables));
401  }
402 
403  }
404  // UPDATE X VALUE
405  for (unsigned int i = 0; i<nodes_.size(); i++)
406  {
407  Map<VectorXs> x_i(nodes_.at(i)->getPtr(), nodes_.at(i)->getStateSize());
408  x_i += x_incr_.segment(nodeLocation(i), nodes_.at(i)->getStateSize());
409  }
410  // Zero the error
411  b_.setZero();
412 
413  time_solving_ += ((double) clock() - t_solving_) / CLOCKS_PER_SEC;
414  n_new_constraints_ = 0;
415  return 1;
416  }
